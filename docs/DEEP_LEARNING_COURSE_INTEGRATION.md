        self,
        features: np.ndarray
    ) -> np.ndarray:
        """Async prediction wrapper –¥–ª—è PyTorch."""
        
        # Blocking –æ–ø–µ—Ä–∞—Ü–∏—é –≤ executor —á—Ç–æ–±—ã –Ω–µ –±–ª–æ–∫–∏—Ä–æ–≤–∞—Ç—å event loop
        return await asyncio.to_thread(
            self._predict_sync,
            features
        )
    
    def _predict_sync(self, features: np.ndarray) -> np.ndarray:
        """–°–∏–Ω—Ö—Ä–æ–Ω–Ω—ã–π PyTorch inference."""
        with torch.no_grad():
            x = torch.tensor(features, dtype=torch.float32).to(self.device)
            output = self.model(x)
            return output.cpu().numpy()
